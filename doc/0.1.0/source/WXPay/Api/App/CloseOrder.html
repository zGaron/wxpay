namespace WXPay\Api\App;

use WXPay\Api;

class CloseOrder extends Api
{

    public api_url = "https://api.mch.weixin.qq.com/pay/closeorder ";

    public is_secure_pay = false;

    /**
     * 公众账号ID String(32)
     *
     * @var string appid wx8888888888888888
     *
     * 
     * 微信分配的公众账号ID（企业号corpid即为此appId）
     * 
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/app.php?chapter=9_3&index=5
     */
    protected appid;
    /**
     * 商户号 String(32)
     *
     * @var string mch_id 1900000109
     *
     * 
     * 微信支付分配的商户号
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/app.php?chapter=9_3&index=5
     */
    protected mch_id;
    /**
     * 商户订单号 String(32)
     *
     * @var string out_trade_no 1217752501201407033233368018
     *
     * 
     * 商户系统内部的订单号
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/app.php?chapter=9_3&index=5
     */
    protected out_trade_no;
    /**
     * 随机字符串 String(32)
     *
     * @var string nonce_str 5K8264ILTKCH16CQ2502SI8ZNMTM67VS
     *
     * 
     * 商户系统内部的订单号,32个字符内、可包含字母, 其他说明见安全规范
     * 
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/app.php?chapter=9_3&index=5
     */
    protected nonce_str;
    /**
     * 签名 String(32)
     *
     * @var string sign C380BEC2BFD727A4B6845133519F3AD6
     *
     * 
     * 签名，详见签名生成算法
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/app.php?chapter=9_3&index=5
     */
    protected sign;

    public _either_groups = [
    ];

    public _required_properties = [
        "appid",
        "mch_id",
        "out_trade_no",
        "nonce_str",
        "sign"
    ];

    public _optional_properties = [
    ];

}
