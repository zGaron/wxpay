Shorturl

namespace WXPay\Api\Native;

use WXPay\Api;

class Shorturl extends Api
{

    public api_url = "https://api.mch.weixin.qq.com/tools/shorturl";

    public is_secure_pay = false;

    /**
     * 公众账号ID String(32)
     *
     * @var string appid wx8888888888888888
     *
     * 
     * 微信分配的公众账号ID（企业号corpid即为此appId）
     * 
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/native.php?chapter=9_9
     */
    public appid;
    /**
     * 商户号 String(32)
     *
     * @var string mch_id 1900000109
     *
     * 
     * 微信支付分配的商户号
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/native.php?chapter=9_9
     */
    public mch_id;
    /**
     * URL链接 String(512、
     *
     * @var string(512、 long_url weixin：//wxpay/bizpayurl?sign=XXXXX&appid=XXXXX&mch_id=XXXXX&product_id=XXXXXX&time_stamp=XXXXXX&nonce_str=XXXXX
     *
     * 
     * 需要转换的URL，签名用原串，传输需URLencode
     * 
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/native.php?chapter=9_9
     */
    public long_url;
    /**
     * 随机字符串 String(32)
     *
     * @var string nonce_str 5K8264ILTKCH16CQ2502SI8ZNMTM67VS
     *
     * 
     * 随机字符串，不长于32位。推荐随机数生成算法
     * 
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/native.php?chapter=9_9
     */
    public nonce_str;
    /**
     * 签名 String(32)
     *
     * @var string sign C380BEC2BFD727A4B6845133519F3AD6
     *
     * 
     * 签名，详见签名生成算法
     * 
     *
     * @see https://pay.weixin.qq.com/wiki/doc/api/native.php?chapter=9_9
     */
    public sign;

    public _either_groups = [
    ];

    public _required_properties = [
        "appid",
        "mch_id",
        "long_url",
        "nonce_str",
        "sign"
    ];

    public _optional_properties = [
    ];

}
